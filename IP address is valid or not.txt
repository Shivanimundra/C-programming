#include<stdio.h>
#include <string.h>
int count=0;
int a=0,b=0,c=0;
int main() {
   char string[50] = "226.300.32.2";
   // Extract the first token
   char * token = strtok(string, ".");
   while( token != NULL )
   {
       
       switch(strlen(token))
       {
          case 1:
                    c= (token[2]-48);
                    break;
                
          case 2:
                    b = (token[1]-48)*10;
                    c= b+(token[2]-48);
                    break;
          case 3:
                    a = (token[0]-48)*100;
                    b = a+(token[1]-48)*10;
                    c= b+(token[2]-48);
                    break;
       }
       if(c<255)
      {
          count++;
          a=b=c=0;
      }
      token = strtok(NULL, ".");
   }
   if(count==4)
   {
     count=0;  
     printf("valid ip address");
   }
   else
   {
       printf("invalid");
   }
   return 0;
}
